package main

import (
	"fmt"
	"sync"
)

type Queue struct {
	sync.Mutex
	data []any
}

func (q *Queue) Put(data any) {
	q.Lock()
	defer q.Unlock()
	q.data = append(q.data, data)
}

func (q *Queue) Get() (any, bool) {
	q.Lock()
	defer q.Unlock()
	if len(q.data) == 0 {
		return nil, false
	}
	data := q.data[0]
	q.data = q.data[1:]
	return data, true
}

func main() {
	q := &Queue{}
	q.Put("123")
	q.Put(nil)
	q.Put(3)

	fmt.Println(q.Get())
	fmt.Println(q.Get())
	fmt.Println(q.Get())
	fmt.Println(q.Get())

}
